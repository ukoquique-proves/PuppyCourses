# ================================================================================
# EXTENSIÓN PARA SCRIPT_GENERATOR.PY (Integración de Comando Click)
# Añade este bloque a tu archivo principal para habilitar el comando de Teledígitos
# ================================================================================

@cli.command('generate-teledigitos')
@click.option('--output', '-o', default='generated_configs/teledigitos_primero.yaml', help='Ruta del archivo YAML de salida.')
@click.option('--ide', default='Cursor', help='IDE con IA que se menciona (Cursor, Trae, Windsurf).')
@click.option('--truco-nombre', default='El Hack del Rendimiento', help='Nombre comercial o gancho del truco.')
def generate_teledigitos(output, ide, truco_nombre):
    """Genera el guion original fusionado con el formato y la marca de Teledígitos."""
    try:
        # 1. Preparar el contexto dinámico para Jinja2
        context = {
            'ide': ide,
            'truco_nombre': truco_nombre
        }
        
        # 2. Cargar y renderizar la plantilla dedicada de Teledígitos
        template = env.get_template('teledigitos_hack.yaml.j2')
        rendered_yaml = template.render(context)
        
        # 3. Validar y parsear el string renderizado para asegurar un formato YAML limpio
        data = yaml.safe_load(rendered_yaml)
        
        # 4. Asegurar que el directorio de salida existe
        out_path = Path(output)
        out_path.parent.mkdir(parents=True, exist_ok=True)
        
        # 5. Guardar el archivo final listo para VideoCreation
        with open(out_path, 'w', encoding='utf-8') as f:
            yaml.dump(data, f, allow_unicode=True, sort_keys=False)
            
        click.echo(f"🚀 ¡Guion de Teledígitos (Fase 2) generado con éxito en: {out_path}!")
        click.echo(f"🎥 Formato: Vertical | Canal: Teledígitos | IDE: {ide}")
        
    except Exception as e:
        click.echo(f"❌ Error al generar el guion de Teledígitos: {e}", err=True)
